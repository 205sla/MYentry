끝 = len(숫자들)
찾음 = 0
while 시작 <= 끝 and 찾음 == 0:
    중간 = (시작 + 끝) // 2
    # 엔트리 블록 표기의 "중간번째 항목" = Python 0-based [중간-1]
    if 숫자들[중간 - 1] == 찾는값:
        찾음 = 중간
    elif 숫자들[중간 - 1] < 찾는값:
        시작 = 중간 + 1
    else:
        끝 = 중간 - 1
