현위치 = 1
틀림 = 0
while 현위치 <= len(괄호) and 틀림 == 0:
    현재 = 괄호[현위치 - 1]
    if 현재 == "(" or 현재 == "[" or 현재 == "{":
        스택.append(현재)
    else:
        if len(스택) == 0:
            틀림 = 1
        else:
            짝 = 스택[-1]
            if (현재 == ")" and 짝 == "(") or (현재 == "]" and 짝 == "[") or (현재 == "}" and 짝 == "{"):
                스택.pop()
            else:
                틀림 = 1
    현위치 = 현위치 + 1
if 틀림 == 1 or len(스택) > 0:
    Entry.print("틀림")
else:
    Entry.print("올바름")
