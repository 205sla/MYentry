# 1단계: 길이 리스트를 [1,1,...,1]로 초기화
for _ in range(len(숫자들)):
    길이.append(1)
# 2단계: DP — 각 i에서 끝나는 LIS 길이
for i in range(2, len(숫자들) + 1):
    for j in range(1, i):
        if 숫자들[j - 1] < 숫자들[i - 1] and 길이[j - 1] + 1 > 길이[i - 1]:
            길이[i - 1] = 길이[j - 1] + 1
# 3단계: 최댓값
최댓값 = 길이[0]
for i in range(1, len(길이)):
    if 길이[i] > 최댓값:
        최댓값 = 길이[i]
Entry.print(최댓값)
